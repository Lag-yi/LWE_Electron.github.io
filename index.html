第一个较大型物品：战神徽章 触发代码

public static void register() {
        ServerLivingEntityEvents.ALLOW_DEATH.register((entity, source, amount) -> {
            // 只在服务端且是玩家
            if (entity.getWorld().isClient || !(entity instanceof PlayerEntity player)) {
                return true;
            }

            // 检查主手或副手是否有战神徽章
            ItemStack mainHand = player.getMainHandStack();
            ItemStack offHand = player.getOffHandStack();
            boolean hasEmblem = mainHand.isOf(ModItems.WARGOD_EMBLEM) ||
                                offHand.isOf(ModItems.WARGOD_EMBLEM);

            if (hasEmblem) {
                // 恢复生命值
                player.setHealth(1.0F);
                // 清除所有负面效果
                player.clearStatusEffects();

                int duration = 800; // 40秒
                // 添加所有增益效果
                player.addStatusEffect(new StatusEffectInstance(StatusEffects.HASTE, duration, 1));
                player.addStatusEffect(new StatusEffectInstance(StatusEffects.SPEED, duration, 1));
                player.addStatusEffect(new StatusEffectInstance(StatusEffects.STRENGTH, duration, 1));
                player.addStatusEffect(new StatusEffectInstance(StatusEffects.JUMP_BOOST, duration, 1));
                player.addStatusEffect(new StatusEffectInstance(StatusEffects.RESISTANCE, duration, 1));
                player.addStatusEffect(new StatusEffectInstance(StatusEffects.REGENERATION, duration, 1));
                player.addStatusEffect(new StatusEffectInstance(StatusEffects.FIRE_RESISTANCE, duration, 1));
                player.addStatusEffect(new StatusEffectInstance(StatusEffects.NIGHT_VISION, duration, 1));
                player.addStatusEffect(new StatusEffectInstance(StatusEffects.HEALTH_BOOST, duration, 1));
                player.addStatusEffect(new StatusEffectInstance(StatusEffects.ABSORPTION, duration, 1));
                player.addStatusEffect(new StatusEffectInstance(StatusEffects.SATURATION, duration, 1));
                player.addStatusEffect(new StatusEffectInstance(StatusEffects.SLOW_FALLING, duration, 1));
                player.addStatusEffect(new StatusEffectInstance(StatusEffects.CONDUIT_POWER, duration, 1));
                player.addStatusEffect(new StatusEffectInstance(StatusEffects.INVISIBILITY, 1200, 1)); // 1分钟隐身

                // 播放音效
                player.getWorld().playSound(null, player.getBlockPos(),
                        SoundEvents.ITEM_TOTEM_USE, SoundCategory.PLAYERS, 1.0F, 1.0F);

                // 消耗徽章
                if (mainHand.isOf(ModItems.WARGOD_EMBLEM)) {
                    mainHand.decrement(1);
                } else {
                    offHand.decrement(1);
                }

                return false; // 阻止死亡
            }
            return true;
        });
    }
